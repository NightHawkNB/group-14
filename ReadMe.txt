	git clone https://github.com/NightHawkNB/group-14.git

Step 02 - Go into the local repository folder
	cd group-14

Step 03 - Set remote Repository
	git remote add origin https://github.com/NightHawkNB/group-14.git

Step 04 - Create Branch
	git branch "branch-name"

Step 05 - Switch to created branch
	git checkout "branch-name"

Step 07 - Add file to local repository
